{

	{
		get { return 10; }
	}

		42.ToString();
	}

	void OnMethod()
	{
		MethodValue().ToString();
	}

	void OnParameter(int param)
	{
    	param.ToString();
	}

	void OnLocalVariable()
	{
        int local = 42;
        local.ToString();
	}
   
	void OnDelegate(System.Func<int> del)
	{
        del().ToString();
	}

	void OnField()
	{
    	field.ToString();
	}

	}
